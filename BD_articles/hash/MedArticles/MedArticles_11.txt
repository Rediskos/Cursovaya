                                                                  В статье представлены результаты исследования посвященного выявлению взаимосвязи между особенностями течения аллергической патологии у детей и возрастом первого контакта с аллергенами кошки Введение  По данным исследований кошки являются чрезвычайно популярными домашними питомцами как в мире так и в нашей стране   Широкое распространение кошек приводит к повсеместной высокой концентрации их аллергенов   Роль экспозиции аллергенов кошек в развитии и течении аллергической патологии согласно литературным данным остается спорной Большинство исследователей отмечают протективное воздействие ранней экспозиции аллергенов кошки  В то же время в ряде исследований показано повышение риска развития аллергических заболеваний и снижение контроля над их течением  Целью настоящего исследования явилось выявление взаимосвязи между особенностями течения аллергической патологии и возрастом первого контакта с аллергенами кошкиМатериал и методы  В исследование были включены  детей проживающих в Свердловской области Средний возраст детей составил  года в гендерном составе преобладали мальчики    Все дети наблюдались аллергологом с диагнозом аллергический ринит АР Верификация диагноза и оценка тяжести течения аллергического заболевания были проведены на основании клинических рекомендаций Российской ассоциации аллергологов и клинических иммунологов РААКИ Экспозиция аллергенов кошки была определена на основании анкетных данных о наличии животного в доме и предположительном возрасте первого контакта с кошками На основании анкетных данных о первом контакте с кошками было выделено несколько групп детей я группа   пациентов  имевших пренатальный контакт с кошками я группа   пациентов  имевших контакт с кошками на первом году жизни я группа    пациент  имевший первый контакт с кошками в возрасте  года я группа   пациентов  имевших первый контакт с кошками в возрасте от  до  лет Всем детям было проведено определение уровня общего иммуноглобулина Е Е сывороточного эозинофильного катионного белка цитологический анализ назального секрета бактериологическое исследование мазка со слизистой полости носа Оценка спектра сенсибилизации проводилась на основании выявленных специфических антител методом иммунофлюоресценции на приборе    ШвецияСтатистическая обработка данных проводилась с использованием следующих параметров средняя величина М медиана  среднее квадратичное отклонение  средняя ошибка средней величины  Достоверность различий между сравниваемыми показателями проводилась с использованием метода однофакторного дисперсионного анализа  использованием пакета прикладных программ     Учитывались результаты с достоверным уровнем различия р и менееРезультаты исследования   Изолированное течение АР было выявлено у  пациентов  Аллергический риноконъюнктивит АРК отмечался в  случаев  Сочетанное течение бронхиальной астмы БА и АР было установлено у   детей Атопический дерматит АД в сочетании с АР был подтвержден у   детей У  детей  отмечалось сочетанное поражение дыхательной системы и кожи в форме БА АР и АД АР у всех детей имел круглогодичный характер течения у большинства   была определена легкая степень тяжести ринита Среднетяжелое течение АР было выявлено у   детей У одного ребенка отмечалось тяжелое течение АР В большинстве случаев   отмечалось легкое интермиттирующее или персистирующее течение БА у   было выявлено среднетяжелое течение заболевания тяжелое течение  у     пациентов имели полный контроль над течением БА по результатам анкеты     При контактные уртикарии возникали у  пациентов Относительный уровень эозинофилии в периферической крови обследованных детей медиана и интерквартильный размах составил    У   детей отмечалось повышение уровня эозинофильного катионного белка Повышение содержания общего  выявлено у  детей  У всех  детей  было выявлено наличие специфических  антител  к эпителию и перхоти кошки сывороточная концентрация которых составила    килоединиц аллергена на литр кЕдАл Антитела к эпителию собаки были выявлены у   пациентов медиана и интерквартильный размах составили    кЕдАл У каждого го ребенка   были выявлены нозологической структуры а также тяжести течения аллергической патологии у пациентов  групп значимых различий выявлено не было табл  Средний уровень общего  у пациентов й группы в  раза превышал результатов  Все обследованные нами дети отмечали что впервые встретились с кошками до  лет большинство   имели контакт с кошками до  лет Полученные данные отображают высокую распространенность кошек в Свердловской области что подтверждает литературные данные и способствует ранней экспозиции аллергенов кошки   Выявленная в ходе исследования высокая частота сочетанных форм аллергической патологии соответствует литературным данным   Преобладали легкие формы атопической патологии более половины детей имели терапевтический контроль над течением респираторной патологии При оценке особенностей течения аллергических заболеваний у детей с разным возрастом первого контакта с аллергенами кошки значимых различий в нозологической структуре и тяжести течения патологии установлено не было Выявлена тенденция к большей распространенности БА у детей имевших пренатальный контакт с аллергенами кошки Мы можем предположить что выявленная тенденция обусловлена генетической предрасположенностью включенных в исследование пациентов к развитию аллергической патологии респираторного тракта  Полученные данные требуют дальнейшего уточнения и проведения генетического обследования пациентов У большинства детей было выявлено повышение уровня общего  отмечалось достоверно более высокое значение показателя у детей имевших контакт с аллергенами кошки на  году жизни В то же время отсутствие значимых различий в тяжести течения аллергической патологии и других лабораторных показателях не позволяют нам подтвердить или опровергнуть протективный эффект пренатального контакта с аллергенами кошки При оценке спектра сенсибилизации была выявлена высокая частота поливалентной гиперчувствительности что соответствует литературным данным  Заключение  В настоящее исследование были включены дети пациентов Была выявлена высокая частота сочетанных форм аллергической патологии преобладали легкие формы АР иили АРК БА и АД Большинство детей впервые  контактировали с аллергенами кошки в раннем возрасте Значимых различий в течении аллергической патологии в зависимости от возраста первого контакта с аллергенами кошки выявлено не было Выявлена тенденция к большей распространенности БА у детей
